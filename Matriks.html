Materi 1 : Matriks Dasar#

import numpy as np
import matplotlib.pyplot as matplot
from sklearn import datasets
%matplotlib inline

Matriks 1 Dimensi#

sekelompok data yang disusun dalam satu baris atau satu kolom, yang dapat direpresentasikan dalam bentuk array satu dimensi.

data = np.array([3,5,6])
print(data)
shape = data.shape
print('Ukuran Matriks =',shape)

[3 5 6]
Ukuran Matriks = (3,)

Matriks 2 Dimensi#

sekelompok data yang disusun dalam beberapa baris dan kolom, yang dapat direpresentasikan dalam bentuk array dua dimensi.

data2 = np.array([[3,5,6],[1,3,2]])
print(data2)
# cek jumlah baris dan kolom
shape = data2.shape
print('Ukuran Matriks =',shape)

[[3 5 6]
 [1 3 2]]
Ukuran Matriks = (2, 3)

Membuat isi matriks dengan data = 0#

np.zeros((baris,kolom))#

zerodata = np.zeros((2,3))

 [0. 0. 0.]]

Membuat isi matriks dengan data = 1#

np.ones((baris,kolom))#

onedata = np.ones((2,3))
print(onedata)

 [1. 1. 1.]]

np.linspace((awal,akhir,jumlahdata))#

Operasi Element Wise (Operasi Antar Element)#

Penjumlahan#

data = np.array([3,5,6])
data2 = np.array([[3,5,6],[1,3,2]])
jumlah = data+data2
print("penjumlahan =")
print(jumlah)

penjumlahan =
[[ 6 10 12]
 [ 4  8  8]]

Perkalian#

data = np.array([3,5,6])
data2 = np.array([[3,5,6],[1,3,2]])
kali = data*data2
print("Perkalian =")
print(kali)

Perkalian =
[[ 9 25 36]
 [ 3 15 12]]

Operasi Matriks Wise#

  • Jumlah kolom harus sama dengan baris

  • Jika jumlah kolom dan baris sama maka akan error

Perkalian#

data3 = np.random.rand(3,1)
